# Using mpvc-tui -n "now playing" notifications also populates mpvc-tui.scb scrobling file.
# The command below generates the "now listening" list shown above this code:

awk '/https:/ { gsub($1,""); gsub($1,""); print }' ~/.config/mpvc/mpvc-tui.scb |
    tac | tail -n 10 | awk '{print $1}' | tee now-listening.csv
